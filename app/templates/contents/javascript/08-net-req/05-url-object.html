URL 생성

let appendUrl = new URL('https://javascript.info/profile/admin');
let url = new URL('tester', appendUrl);  // https://javascript.info/profile/tester
url.protocol; //https

  url.pathname; // /profile/admin

protocol, hostname, port

SearchParams : "?..."

new URL('https://google.com/search?query=JavaScript')

append(name, value) – add the parameter by name

delete(name) – remove the parameter by name

get(name) – get the parameter by name

getAll(name) – get all parameters with the same name (that’s possible, e.g. ?user=John&user=Pete)

has(name) – check for the existence of the parameter by name

set(name, value) – set/replace the parameter

sort() – sort parameters by name, rarely needed

…and it’s also iterable, similar to Map.

let url = new URL('https://google.com/search');
url.searchParams.set('q', 'test me!'); // https://google.com/search?q=test+me%21
url.searchParams.set('tbs', 'qdr:y'); // .../search?q=test+me%21&tbs=qdr%3Ay

for(let [name, value] of url.searchParams) {
  alert(`${name}=${value}`); // q=test me!, then tbs=qdr:y
}
