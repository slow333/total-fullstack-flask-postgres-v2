r.random() # 0.12345678901234567

r.randrange(1, 3) # 1부터 3 사이의 임의의 정수(3은 포함하지 않음)
mylist = [1, 2, 3, 4, 5]
r.sample(mylist, 2) # mylist에서 임의의 요소 2개를 뽑아 선택 중복불가
r.choices(mylist, 2) # mylist에서 임의의 요소 2개를 뽑아 선택 중복허용
